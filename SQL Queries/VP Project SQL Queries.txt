CREATE FUNCTION fn_GetReceivablesAgingBucket(@DueDate DATE)
RETURNS NVARCHAR(50)
AS
BEGIN
    DECLARE @DaysOverdue INT = DATEDIFF(DAY, @DueDate, GETDATE());
    RETURN CASE 
        WHEN @DaysOverdue <= 0 THEN 'Not Due'
        WHEN @DaysOverdue <= 30 THEN '0-30 Days'
        WHEN @DaysOverdue <= 60 THEN '31-60 Days'
        WHEN @DaysOverdue <= 90 THEN '61-90 Days'
        ELSE '90+ Days'
    END;
END;
GO

CREATE FUNCTION fn_GetBuyerOutstandingBalance(@BuyerId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    DECLARE @Balance DECIMAL(18,2);
    SELECT @Balance = ISNULL(SUM(inst.Amount - inst.PaidAmount), 0)
    FROM Installments inst
    INNER JOIN PaymentPlans pp ON inst.PaymentPlanId = pp.PaymentPlanId
    INNER JOIN Sales s ON pp.SaleId = s.SaleId
    WHERE s.BuyerId = @BuyerId AND inst.Status IN ('Due','Overdue','Partial');
    RETURN ISNULL(@Balance, 0);
END;
GO

CREATE FUNCTION fn_GetProjectTotalSales(@ProjectId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    DECLARE @TotalSales DECIMAL(18,2);
    SELECT @TotalSales = ISNULL(SUM(SalePrice), 0) FROM Sales WHERE ProjectId = @ProjectId AND Status = 'Completed';
    RETURN ISNULL(@TotalSales, 0);
END;
GO

CREATE FUNCTION fn_GetProjectTotalExpenses(@ProjectId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    DECLARE @TotalExpenses DECIMAL(18,2);
    SELECT @TotalExpenses = ISNULL(SUM(Amount), 0) FROM Expenses WHERE ProjectId = @ProjectId;
    RETURN ISNULL(@TotalExpenses, 0);
END;
GO

CREATE FUNCTION fn_GetProjectNetProfit(@ProjectId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    RETURN ISNULL(dbo.fn_GetProjectTotalSales(@ProjectId) - dbo.fn_GetProjectTotalExpenses(@ProjectId), 0);
END;
GO

CREATE FUNCTION fn_GetAgentTotalCommission(@AgentId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    DECLARE @Total DECIMAL(18,2);
    SELECT @Total = ISNULL(SUM(CommissionAmount), 0) FROM Commissions WHERE AgentId = @AgentId;
    RETURN ISNULL(@Total, 0);
END;
GO

CREATE FUNCTION fn_GetAgentPendingCommission(@AgentId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    DECLARE @Pending DECIMAL(18,2);
    SELECT @Pending = ISNULL(SUM(CommissionAmount), 0) FROM Commissions WHERE AgentId = @AgentId AND Status = 'Pending';
    RETURN ISNULL(@Pending, 0);
END;
GO

CREATE FUNCTION fn_GetPartnerRunningBalance(@PartnerId INT, @ProjectId INT)
RETURNS DECIMAL(18,2)
AS
BEGIN
    DECLARE @Bal DECIMAL(18,2);
    SELECT @Bal = (CapitalInjected + ProfitCredited - Withdrawals)
    FROM PartnerInvestments WHERE PartnerId = @PartnerId AND ProjectId = @ProjectId;
    RETURN ISNULL(@Bal, 0);
END;
GO

CREATE PROCEDURE sp_CreatePaymentPlan
    @SaleId INT,
    @PlanType NVARCHAR(50),
    @TotalAmount DECIMAL(18,2),
    @DownPayment DECIMAL(18,2),
    @NumberOfInstallments INT,
    @StartDate DATE
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @InstallmentAmount DECIMAL(18,2);
    DECLARE @PaymentPlanId INT;
    DECLARE @Counter INT = 1;
    DECLARE @DueDate DATE;

    BEGIN TRANSACTION;
    BEGIN TRY
        IF @NumberOfInstallments <= 0 THROW 50001, 'Number of installments must be greater than 0', 1;
        IF @TotalAmount <= @DownPayment THROW 50002, 'Total amount must be greater than down payment', 1;

        SET @InstallmentAmount = (@TotalAmount - @DownPayment) / @NumberOfInstallments;

        INSERT INTO PaymentPlans (SaleId, PlanType, TotalAmount, DownPayment, NumberOfInstallments, InstallmentAmount, StartDate)
        VALUES (@SaleId, @PlanType, @TotalAmount, @DownPayment, @NumberOfInstallments, @InstallmentAmount, @StartDate);

        SET @PaymentPlanId = SCOPE_IDENTITY();
        UPDATE Sales SET PaymentPlanId = @PaymentPlanId WHERE SaleId = @SaleId;

        SET @DueDate = @StartDate;
        WHILE @Counter <= @NumberOfInstallments
        BEGIN
            IF @PlanType = 'Monthly' SET @DueDate = DATEADD(MONTH, @Counter, @StartDate);
            ELSE IF @PlanType = 'Quarterly' SET @DueDate = DATEADD(MONTH, @Counter * 3, @StartDate);
            ELSE IF @PlanType = 'BiAnnual' SET @DueDate = DATEADD(MONTH, @Counter * 6, @StartDate);
            ELSE SET @DueDate = DATEADD(MONTH, @Counter, @StartDate);

            INSERT INTO Installments (PaymentPlanId, InstallmentNo, DueDate, Amount, Status)
            VALUES (@PaymentPlanId, @Counter, @DueDate, @InstallmentAmount, 'Due');
            SET @Counter += 1;
        END

        COMMIT TRANSACTION;
        SELECT @PaymentPlanId AS PaymentPlanId, 'Payment plan created successfully' AS Message;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO

CREATE PROCEDURE sp_RecordPayment
    @InstallmentId INT,
    @Amount DECIMAL(18,2),
    @PaymentDate DATE,
    @AccountId INT,
    @Method NVARCHAR(50),
    @Description NVARCHAR(500) = NULL
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @SaleId INT;
    DECLARE @BuyerId INT;
    DECLARE @TransactionId INT;
    DECLARE @InstallmentAmount DECIMAL(18,2);
    DECLARE @CurrentPaidAmount DECIMAL(18,2);

    BEGIN TRANSACTION;
    BEGIN TRY
        SELECT @SaleId = s.SaleId, @BuyerId = s.BuyerId,
               @InstallmentAmount = inst.Amount, @CurrentPaidAmount = inst.PaidAmount
        FROM Installments inst
        INNER JOIN PaymentPlans pp ON inst.PaymentPlanId = pp.PaymentPlanId
        INNER JOIN Sales s ON pp.SaleId = s.SaleId
        WHERE inst.InstallmentId = @InstallmentId;

        IF @SaleId IS NULL THROW 50003, 'Installment not found', 1;
        IF @Amount <= 0 THROW 50004, 'Payment amount must be greater than 0', 1;

        INSERT INTO Transactions (Date, Amount, Type, Reference, ToPartyId, SaleId, AccountId, InstallmentId, Method, Description)
        VALUES (@PaymentDate, @Amount, 'Payment',
                'Installment Payment - Installment #' + CAST((SELECT InstallmentNo FROM Installments WHERE InstallmentId = @InstallmentId) AS NVARCHAR),
                @BuyerId, @SaleId, @AccountId, @InstallmentId, @Method, @Description);

        SET @TransactionId = SCOPE_IDENTITY();

        UPDATE Installments
        SET PaidAmount = PaidAmount + @Amount,
            Status = CASE 
                WHEN (PaidAmount + @Amount) >= Amount THEN 'Paid'
                WHEN (PaidAmount + @Amount) > 0 THEN 'Partial'
                ELSE 'Due'
            END,
            PaidDate = CASE WHEN (PaidAmount + @Amount) >= Amount THEN @PaymentDate ELSE PaidDate END,
            TransactionId = @TransactionId,
            UpdatedAt = GETDATE()
        WHERE InstallmentId = @InstallmentId;

        COMMIT TRANSACTION;
        SELECT @TransactionId AS TransactionId, 'Payment recorded successfully' AS Message;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO

CREATE PROCEDURE sp_DistributeProfit
    @ProjectId INT,
    @DistributionDate DATE,
    @TotalProfit DECIMAL(18,2),
    @CreatedBy INT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @DistributionId INT;
    DECLARE @PartnerInvestmentId INT;
    DECLARE @SharePercent DECIMAL(5,2);
    DECLARE @ProfitAmount DECIMAL(18,2);

    BEGIN TRANSACTION;
    BEGIN TRY
        IF @TotalProfit <= 0 THROW 50005, 'Total profit must be greater than 0', 1;

        INSERT INTO ProfitDistributions (ProjectId, DistributionDate, TotalProfit, CreatedBy, Status)
        VALUES (@ProjectId, @DistributionDate, @TotalProfit, @CreatedBy, 'Pending');

        SET @DistributionId = SCOPE_IDENTITY();

        DECLARE partner_cursor CURSOR FOR
        SELECT PartnerInvestmentId, SharePercent
        FROM PartnerInvestments
        WHERE ProjectId = @ProjectId AND Status = 'Active';

        OPEN partner_cursor;
        FETCH NEXT FROM partner_cursor INTO @PartnerInvestmentId, @SharePercent;
        WHILE @@FETCH_STATUS = 0
        BEGIN
            SET @ProfitAmount = @TotalProfit * @SharePercent / 100;
            INSERT INTO ProfitDistributionDetails (DistributionId, PartnerInvestmentId, SharePercent, ProfitAmount, Status)
            VALUES (@DistributionId, @PartnerInvestmentId, @SharePercent, @ProfitAmount, 'Pending');
            FETCH NEXT FROM partner_cursor INTO @PartnerInvestmentId, @SharePercent;
        END
        CLOSE partner_cursor;
        DEALLOCATE partner_cursor;

        UPDATE ProfitDistributions SET Status = 'Distributed' WHERE DistributionId = @DistributionId;

        COMMIT TRANSACTION;
        SELECT @DistributionId AS DistributionId, 'Profit distributed successfully' AS Message;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO

CREATE PROCEDURE sp_MarkPlotSold
    @PlotId INT,
    @BuyerId INT,
    @SalePrice DECIMAL(18,2),
    @SaleDate DATE,
    @DownPayment DECIMAL(18,2),
    @SellerId INT,
    @AgentId INT = NULL,
    @CommissionPercent DECIMAL(5,2) = NULL,
    @CreatedBy INT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @SaleId INT;
    DECLARE @ProjectId INT;

    BEGIN TRANSACTION;
    BEGIN TRY
        SELECT @ProjectId = ProjectId FROM Plots WHERE PlotId = @PlotId;
        IF @ProjectId IS NULL THROW 50006, 'Plot not found', 1;
        IF EXISTS (SELECT 1 FROM Plots WHERE PlotId = @PlotId AND Status = 'Sold') THROW 50007, 'Plot already sold', 1;

        INSERT INTO Sales (ProjectId, PlotId, BuyerId, SellerId, SalePrice, SaleDate, DownPayment, AgentId, CommissionPercent, CreatedBy, Status)
        VALUES (@ProjectId, @PlotId, @BuyerId, @SellerId, @SalePrice, @SaleDate, @DownPayment, @AgentId, @CommissionPercent, @CreatedBy, 'Pending');

        SET @SaleId = SCOPE_IDENTITY();
        UPDATE Plots SET Status = 'Sold', BuyerId = @BuyerId, UpdatedAt = GETDATE() WHERE PlotId = @PlotId;

        COMMIT TRANSACTION;
        SELECT @SaleId AS SaleId, 'Plot marked as sold successfully' AS Message;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO

CREATE PROCEDURE sp_RecordCommissionPayment
    @CommissionId INT,
    @PaymentDate DATE,
    @AccountId INT,
    @Method NVARCHAR(50),
    @ApprovedBy INT,
    @Description NVARCHAR(500) = NULL
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @TransactionId INT;
    DECLARE @AgentId INT;
    DECLARE @Amount DECIMAL(18,2);

    BEGIN TRANSACTION;
    BEGIN TRY
        SELECT @AgentId = AgentId, @Amount = CommissionAmount FROM Commissions WHERE CommissionId = @CommissionId;
        IF @AgentId IS NULL THROW 50010, 'Commission not found', 1;

        INSERT INTO Transactions (Date, Amount, Type, Reference, ToPartyId, AccountId, Method, Description)
        VALUES (@PaymentDate, @Amount, 'Commission', 'Commission Payment - Commission #' + CAST(@CommissionId AS NVARCHAR), @AgentId, @AccountId, @Method, @Description);

        SET @TransactionId = SCOPE_IDENTITY();

        UPDATE Commissions
        SET Status = 'Paid', PaymentVoucherId = @TransactionId, ApprovedBy = @ApprovedBy,
            ApprovedDate = @PaymentDate, PaidDate = @PaymentDate, UpdatedAt = GETDATE()
        WHERE CommissionId = @CommissionId;

        COMMIT TRANSACTION;
        SELECT @TransactionId AS TransactionId, 'Commission payment recorded successfully' AS Message;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO

CREATE VIEW vw_SalesRegister AS
SELECT 
    s.SaleId,
    p.Name AS ProjectName,
    pl.PlotNo,
    pl.SizeMarla,
    pl.Block,
    pl.Phase,
    buyer.Name AS BuyerName,
    buyer.CNIC AS BuyerCNIC,
    buyer.ContactPhone AS BuyerPhone,
    seller.Name AS SellerName,
    s.SalePrice,
    s.DownPayment,
    s.SaleDate,
    pp.PlanType,
    pp.TotalAmount,
    pp.NumberOfInstallments,
    ISNULL(SUM(inst.PaidAmount), 0) AS TotalPaid,
    (pp.TotalAmount - ISNULL(SUM(inst.PaidAmount), 0)) AS BalanceReceivable,
    s.Status AS SaleStatus
FROM Sales s
INNER JOIN Projects p ON s.ProjectId = p.ProjectId
INNER JOIN Plots pl ON s.PlotId = pl.PlotId
INNER JOIN Parties buyer ON s.BuyerId = buyer.PartyId
INNER JOIN Parties seller ON s.SellerId = seller.PartyId
LEFT JOIN PaymentPlans pp ON s.PaymentPlanId = pp.PaymentPlanId
LEFT JOIN Installments inst ON pp.PaymentPlanId = inst.PaymentPlanId
GROUP BY s.SaleId, p.Name, pl.PlotNo, pl.SizeMarla, pl.Block, pl.Phase,
    buyer.Name, buyer.CNIC, buyer.ContactPhone, seller.Name,
    s.SalePrice, s.DownPayment, s.SaleDate, pp.PlanType, pp.TotalAmount,
    pp.NumberOfInstallments, s.Status;
GO

CREATE VIEW vw_ReceivablesAging AS
SELECT 
    s.SaleId,
    buyer.Name AS BuyerName,
    buyer.CNIC AS BuyerCNIC,

CREATE VIEW vw_BuyerLedger AS
SELECT 
    buyer.PartyId,
    buyer.Name AS BuyerName,
    buyer.CNIC,
    buyer.ContactPhone,
    buyer.ContactEmail,
    t.TransactionId,
    t.Date AS TransactionDate,

    CASE WHEN t.Type = 'Refund' THEN t.Amount ELSE 0 END AS Credit

CREATE VIEW vw_MonthlyProgressReport AS
SELECT 
    YEAR(s.SaleDate) AS Year,
    MONTH(s.SaleDate) AS Month,
    p.ProjectId,
    p.Name AS ProjectName,
    COUNT(DISTINCT s.SaleId) AS TotalSales,
    SUM(s.SalePrice) AS TotalSalesValue,
    SUM(s.DownPayment) AS TotalDownPayments,
    COUNT(DISTINCT pl.PlotId) AS NewPlotsOpened,
    ISNULL(SUM(exp.Amount), 0) AS TotalExpenses,
    (SUM(s.SalePrice) - ISNULL(SUM(exp.Amount), 0)) AS NetProfit
FROM Sales s
INNER JOIN Projects p ON s.ProjectId = p.ProjectId
INNER JOIN Plots pl ON s.PlotId = pl.PlotId
LEFT JOIN Expenses exp ON p.ProjectId = exp.ProjectId 
    AND YEAR(exp.ExpenseDate) = YEAR(s.SaleDate) 
    AND MONTH(exp.ExpenseDate) = MONTH(s.SaleDate)
GROUP BY YEAR(s.SaleDate), MONTH(s.SaleDate), p.ProjectId, p.Name;
GO

CREATE TRIGGER TR_Sales_UpdatePlotStatus ON Sales AFTER UPDATE AS
BEGIN SET NOCOUNT ON;
    IF UPDATE(Status)
    BEGIN
        UPDATE Plots SET Status = 'Sold', BuyerId = i.BuyerId, UpdatedAt = GETDATE()
        FROM Plots p INNER JOIN inserted i ON p.PlotId = i.PlotId
        WHERE i.Status = 'Completed'
          AND EXISTS (SELECT 1 FROM deleted d WHERE d.SaleId = i.SaleId AND d.Status <> 'Completed');
    END

CREATE TRIGGER TR_Transactions_UpdateInstallment ON Transactions AFTER INSERT AS
BEGIN SET NOCOUNT ON;
    UPDATE Installments
    SET PaidAmount = PaidAmount + i.Amount,
        Status = CASE 
            WHEN (PaidAmount + i.Amount) >= Amount THEN 'Paid'
            WHEN (PaidAmount + i.Amount) > 0 THEN 'Partial'
            ELSE Status
        END,
        PaidDate = CASE WHEN (PaidAmount + i.Amount) >= Amount THEN i.Date ELSE PaidDate END,
        TransactionId = i.TransactionId,
        UpdatedAt = GETDATE()
    FROM Installments inst INNER JOIN inserted i ON inst.InstallmentId = i.InstallmentId
    WHERE i.InstallmentId IS NOT NULL;
END;
GO

CREATE TRIGGER TR_Sales_AutoCalculateCommission ON Sales AFTER INSERT, UPDATE AS
BEGIN SET NOCOUNT ON;
    INSERT INTO Commissions (SaleId, AgentId, CommissionPercent, CommissionAmount, Status)
    SELECT i.SaleId, i.AgentId, i.CommissionPercent, (i.SalePrice * i.CommissionPercent / 100), 'Pending'
    FROM inserted i
    WHERE i.AgentId IS NOT NULL AND i.CommissionPercent IS NOT NULL AND i.CommissionPercent > 0
      AND NOT EXISTS (SELECT 1 FROM Commissions c WHERE c.SaleId = i.SaleId AND c.AgentId = i.AgentId);

    UPDATE c SET CommissionAmount = (i.SalePrice * i.CommissionPercent / 100),
                 CommissionPercent = i.CommissionPercent, UpdatedAt = GETDATE()
    FROM Commissions c INNER JOIN inserted i ON c.SaleId = i.SaleId
    WHERE i.AgentId IS NOT NULL AND i.CommissionPercent IS NOT NULL AND c.Status = 'Pending';
END;
GO

CREATE TRIGGER TR_Transactions_UpdateBankBalance ON Transactions AFTER INSERT, UPDATE, DELETE AS
BEGIN SET NOCOUNT ON;
    IF EXISTS (SELECT 1 FROM inserted)
    BEGIN
        UPDATE BankAccounts
        SET Balance = Balance + CASE 
                WHEN i.Type IN ('Payment','Commission','SellerPayment') THEN i.Amount
                WHEN i.Type IN ('Withdrawal','PartnerWithdrawal','Refund') THEN -i.Amount
                ELSE 0
            END, UpdatedAt = GETDATE()
        FROM BankAccounts ba INNER JOIN inserted i ON ba.AccountId = i.AccountId
        WHERE i.AccountId IS NOT NULL;
    END

    IF EXISTS (SELECT 1 FROM deleted)
    BEGIN
        UPDATE BankAccounts
        SET Balance = Balance - CASE 
                WHEN d.Type IN ('Payment','Commission','SellerPayment') THEN d.Amount
                WHEN d.Type IN ('Withdrawal','PartnerWithdrawal','Refund') THEN -d.Amount
                ELSE 0
            END, UpdatedAt = GETDATE()
        FROM BankAccounts ba INNER JOIN deleted d ON ba.AccountId = d.AccountId
        WHERE d.AccountId IS NOT NULL;
    END
END;
